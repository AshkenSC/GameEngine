#version 330

in vec2 pass_textureCoords;

out vec4 out_colour;

uniform vec3 colour;
uniform sampler2D fontAtlas;

// for smaller texts, use higher edge values and lower width values
// for larger texts, use lower edge values and higher width values
// to avoid blurry edges and get better display effect
const float width = 0.5;
const float edge = 0.1;

void main(void) {

	float distance = 1.0 - texture(fontAtlas, pass_textureCoords).a;
	float alpha = 1.0 - smoothstep(width, width + edge, distance);
	
	out_colour = vec4(colour, alpha);
	
}
